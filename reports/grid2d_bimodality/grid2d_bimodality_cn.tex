\section{模型定义与坐标约定}
状态空间为 $\{1,\dots,N\}\times\{1,\dots,N\}$。记当前位置为 $(x,y)$，目标点 $(x_t,y_t)$ 为吸收态。每步更新为
\begin{align}
  p_{\text{left}} &= \frac{q}{4}(1+g_x),\quad p_{\text{right}} = \frac{q}{4}(1-g_x),\\
  p_{\text{down}} &= \frac{q}{4}(1+g_y),\quad p_{\text{up}} = \frac{q}{4}(1-g_y),\\
  p_{\text{stay}} &= 1-q.
\end{align}
本文采用 ``$y$ 轴向下为正'' 的坐标约定，因此 $g_y>0$ 表示向下偏置。边界条件支持反射、周期或混合（$x$ 周期、$y$ 反射）。
\noindent 由式(1)可见 \(g_x>0\) 会增大 \(p_{\text{left}}=\frac{q}{4}(1+g_x)\) 并减小 \(p_{\text{right}}=\frac{q}{4}(1-g_x)\)，因此本文约定 \(g_x>0\) 表示\textbf{向左}全局偏置（\(g_x<0\) 表示向右偏置）。

\paragraph{局部异质性（组合）}
\begin{itemize}
  \item local bias（箭头）：将 $p_{\text{stay}}$ 乘 $(1-\delta)$，并把 $\Delta=\delta p_{\text{stay}}$ 加到箭头方向；其余方向不变。
  \item sticky site：将该格点的移动概率缩放为 $q_{\text{site}}=\text{factor}\cdot q$（本文默认 0.2），$p_{\text{stay}}$ 相应上升。
  \item barrier：反射边将该方向概率加回 stay；半透门以 $p_{\text{pass}}$ 通过、$1-p_{\text{pass}}$ 反射。
\end{itemize}

\section{解析推导：从无缺陷 propagator 到时域 FPT}
本节给出统一推导链条：\emph{无缺陷 propagator $\to$ 有缺陷 propagator $\to$ FPT 生成函数 $\to$ AW 反演}。三类候选均可落在同一框架内。

\subsection{矩阵形式与生成函数}
设无吸收系统的转移矩阵为 $A$，约定 $A_{v,u}=\Pr(u\to v)$，因此列和为 1，且 $\mathbf{p}_{t+1}=A\mathbf{p}_t$。传播子满足
\begin{equation}
  P_{n_0}(n,t)=\mathbf{e}_n^\top A^t \mathbf{e}_{n_0},
  \qquad
  \tilde{\mathbf{P}}(z)=\sum_{t\ge 0} A^t z^t=(I-zA)^{-1},
\end{equation}
从而
\begin{equation}
  \tilde P_{n_0}(n,z)=\mathbf{e}_n^\top (I-zA)^{-1}\mathbf{e}_{n_0}.
\end{equation}
下面给出 $A$ 的谱分解形式与缺陷更新公式，以便对任意候选构造得到 $\tilde P$。

\subsection{无缺陷 propagator 的谱分解}
将二维转移写成两个一维算子叠加（沿 $x$ 与 $y$ 方向）。设一维步进矩阵的特征对为 $\{\lambda_k^{(\gamma)},h_k^{(\gamma)}\}$，其中 $\gamma\in\{\text{periodic},\text{reflecting}\}$ 表示边界类型。定义
\begin{equation}
  s_k^{(\gamma)} = q\,\bigl(\lambda_k^{(\gamma)}-1\bigr),\qquad
  \Lambda_{k_1,k_2}=1+\frac{s_{k_1}^{(\gamma_x)}+s_{k_2}^{(\gamma_y)}}{2}.
\end{equation}
则二维无缺陷 propagator（PRE 102, 062124 的 Eq.(33) 形式）为
\begin{equation}
  P^{(\gamma)}_{n_0}(n,t)=\sum_{k_1}\sum_{k_2} h_{k_1}^{(\gamma_x)}(x,x_0)\,h_{k_2}^{(\gamma_y)}(y,y_0)\,\Lambda_{k_1,k_2}^t.
  \label{eq:P_time}
\end{equation}
对时间做 $z$ 变换得到
\begin{equation}
  \tilde P^{(\gamma)}_{n_0}(n,z)=\sum_{k_1}\sum_{k_2}
  \frac{h_{k_1}^{(\gamma_x)}(x,x_0)\,h_{k_2}^{(\gamma_y)}(y,y_0)}{1-z\Lambda_{k_1,k_2}}.
  \label{eq:P_z}
\end{equation}

\paragraph{周期边界（periodic）} \;\;
\begin{equation}
  \theta_k=\frac{2\pi k}{N},\quad
  \lambda_k^{(\text{per})}=\frac{1+g}{2}e^{-i\theta_k}+\frac{1-g}{2}e^{i\theta_k},\quad
  h_k^{(\text{per})}(x,x_0)=\frac{1}{N}e^{i\theta_k(x-x_0)}.
\end{equation}

\paragraph{反射边界（reflecting）} \;\;
设 $p_\ell=(1+g)/2,\;p_r=(1-g)/2,\;\rho=p_r/p_\ell$。可通过相似变换将一维算子对称化，得到
\begin{equation}
  \lambda_0^{(\text{ref})}=1,\quad
  \lambda_k^{(\text{ref})}=2\sqrt{p_\ell p_r}\cos\frac{k\pi}{N}\ (k\ge 1),\quad
  h_k^{(\text{ref})}(x,x_0)=u_k(x)u_k(x_0)\sqrt{\frac{\pi_x}{\pi_{x_0}}},
\end{equation}
其中 $\pi_x\propto \rho^x$ 为平衡分布，$u_k$ 为正交余弦模（与 PRE Eq.(22)-(23) 一致）。

\subsection{有缺陷 propagator 的 determinant 公式}
将缺陷写为矩阵扰动 $\underline{X}=\underline{B}-\underline{A}$。对每条有向边 $(u_i\to v_i)$ 定义
\begin{equation}
  \eta_{u_i,v_i}=X_{v_i,u_i}=B_{v_i,u_i}-A_{v_i,u_i}.
\end{equation}
并定义差分 propagator
\begin{equation}
  \tilde P^e_{\langle u-v\rangle}(n,z)=\tilde P^e_{u}(n,z)-\tilde P^e_{v}(n,z).
\end{equation}
则 2311.00464v2 的 inert-heterogeneity 公式给出
\begin{equation}
  \tilde P_{n_0}(n,z)=\tilde P^e_{n_0}(n,z)-1+\frac{\det[H(n,n_0;z)]}{\det[H(z)]},
\end{equation}
其中
\begin{align}
  H_{ij}(z)&=-\frac{\delta_{ij}}{z}+\eta_{u_j,v_j}\,\tilde P^e_{\langle u_j-v_j\rangle}(u_i,z)-\eta_{v_j,u_j}\,\tilde P^e_{\langle u_j-v_j\rangle}(v_i,z),\\
  H(n,n_0;z)_{ij}&=H_{ij}(z)-\tilde P^e_{\langle u_j-v_j\rangle}(n,z)\left[\eta_{u_i,v_i}\tilde P^e_{n_0}(u_i,z)-\eta_{v_i,u_i}\tilde P^e_{n_0}(v_i,z)\right].
\end{align}
数值实现中，我们等价地求解线性系统得到 $\tilde P$（由 Cramer 规则与上式等价），以获得更好的数值稳定性。
亦可将其写成线性代数形式：
\begin{align}
  b_j(n,z)&=\tilde P^e_{u_j}(n,z)-\tilde P^e_{v_j}(n,z),\\
  a_i(n_0,z)&=\eta_{u_i,v_i}\tilde P^e_{n_0}(u_i,z)-\eta_{v_i,u_i}\tilde P^e_{n_0}(v_i,z),\\
  H_{ij}(z)&=-\frac{\delta_{ij}}{z}+\eta_{u_j,v_j}\,b_j(u_i,z)-\eta_{v_j,u_j}\,b_j(v_i,z),
\end{align}
从而
\begin{equation}
  \tilde P_{n_0}(n,z)=\tilde P^e_{n_0}(n,z)-\mathbf{b}(n,z)^\top H(z)^{-1}\mathbf{a}(n_0,z).
\end{equation}
本文数值实现使用该线性系统形式避免逐点求行列式。

\paragraph{缺陷映射（B/C 使用）}
\begin{itemize}
  \item local bias：减少 self-loop 并把概率加到箭头方向，等价于 $(u\to v)$ 的 $\eta_{u,v}=+\Delta$，并对 $(u\to u)$ 添加补偿项以保持列和为 1。
  \item door/permeable barrier：跨门通过概率乘 $p_{\text{pass}}$，对应边 $(u\to v)$ 与 $(v\to u)$ 具有 $\eta_{u,v}=(p_{\text{pass}}-1)P_{u\to v}^{\text{base}}$。
  \item sticky：所有出边概率乘 $\alpha$，每条出边 $(u\to n)$ 的 $\eta_{u,n}=(\alpha-1)P_{u\to n}^{\text{base}}$，并对 $(u\to u)$ 添加补偿项以保持列和为 1。
\end{itemize}

\subsection{FPT 生成函数与 AW 反演}
考虑包含吸收态的主方程
\begin{equation}
  \mathbf{p}_{t+1}=P\,\mathbf{p}_t,\qquad \mathbf{p}_0=\mathbf{e}_{n_0},
\end{equation}
对时间做 $z$ 变换 $\tilde{\mathbf{p}}(z)=\sum_{t\ge 0}\mathbf{p}_t z^t$ 得
\begin{equation}
  \tilde{\mathbf{p}}(z)=(I-zP)^{-1}\mathbf{p}_0.
\end{equation}
记累积到达概率 $F(t)=\mathbf{e}_{n_t}^\top \mathbf{p}_t$，则
\begin{equation}
  \tilde F(z)=\mathbf{e}_{n_t}^\top (I-zP)^{-1}\mathbf{p}_0.
\end{equation}
由 $f(t)=F(t)-F(t-1)$（取 $F(-1)=0$）得到 PMF 的生成函数
\begin{equation}
  \tilde f(z)=(1-z)\tilde F(z)=(1-z)\,\mathbf{e}_{n_t}^\top (I-zP)^{-1}\mathbf{p}_0.
\end{equation}
本文保留两者记号：$\tilde F$ 为 CDF 的生成函数，$\tilde f$ 为 PMF 的生成函数（AW 反演使用 $\tilde f$）。
Exact recursion 的时域表达为
\begin{equation}
  f(t)=r^\top Q^{t-1} p_0,\quad t\ge 1,
  \label{eq:exact-recursion}
\end{equation}
因此生成函数可写为
\begin{equation}
  \tilde f(z)=\sum_{t\ge 1} f(t) z^t
  = z\,r^\top (I-zQ)^{-1} p_0.
\end{equation}
另一方面，由 renewal 关系亦有
\begin{equation}
  \tilde f_{n_0\to n_t}(z)=\frac{\tilde P_{n_0}(n_t,z)}{\tilde P_{n_t}(n_t,z)},
  \label{eq:Fz}
\end{equation}
两者等价：前者以 transient 子矩阵 $Q$ 表达，后者以无吸收 propagator 表达。
随后采用 Abate--Whitt (AW) 数值反演将 $z$ 域系数化：
\begin{equation}
  f(t)\approx r^{-t}\frac{1}{m}\sum_{k=0}^{m-1}\tilde f(z_k)\,e^{-2\pi i k t/m},
  \qquad z_k=r\,e^{2\pi i k/m},\quad t=1,\dots,t_{\max}^{\mathrm{AW}}.
  \label{eq:AW}
\end{equation}
本文选取 $m=\mathrm{pow2}(\text{oversample}\cdot(t_{\max}^{\mathrm{AW}}+1))$，并设 $r^m=10^{-\text{r\_pow10}}$，与仓库内 AW 设置保持一致。
其中 $r$ 越接近 1，截断误差越小但数值更敏感；$r$ 过小则会放大截断误差，本文沿用自动选参策略以平衡稳定性与精度。

\begin{quote}
\textbf{实现流程（Algorithm box）}
\begin{enumerate}[leftmargin=*, itemsep=0pt]
  \item 构建无缺陷 $A$ 的谱分解（周期/反射边界），得到 $\tilde P^e(n,z)$ 的显式模态和。
  \item 由缺陷列表构造 $H(z)$ 与向量 $\mathbf{a},\mathbf{b}$，解线性系统得到 $\tilde P(n,z)$。
  \item 用 renewal 公式 $\tilde f(z)=\tilde P_{n_0}(n_t,z)/\tilde P_{n_t}(n_t,z)$ 得到生成函数。
  \item 对 $z_k=r e^{2\pi i k/m}$ 采样并用 AW 反演得到 $f(t)$。
  \item 与 exact recursion/MC 对比误差，并记录为指标 JSON。
\end{enumerate}
\end{quote}

\section{数值基线：Exact/flux 递推与 MC}
Exact 递推在 transient 空间计算 $f(t)=r^\top Q^{t-1} p_0$（式~\ref{eq:exact-recursion}），作为确定性基准，并与 AW 反演及 MC 直方图叠加对比（图~\ref{fig:A-fpt}--\ref{fig:C-fpt}）。MC 轨迹同时按 ``快/慢通道'' 分类，以解释双峰来源。

\paragraph{峰谷指标定义}
对平滑后的 $f(t)$，取两个局部主峰位置 $t_{p1},t_{p2}$ 及其间最小值 $t_v$，并定义
\begin{equation}
  h_1=f(t_{p1}),\quad h_2=f(t_{p2}),\quad
  \text{peak ratio}=\frac{\min(h_1,h_2)}{\max(h_1,h_2)},\quad
  \text{valley ratio}=\frac{f(t_v)}{\min(h_1,h_2)}.
\end{equation}
% --- clarification: v_max vs v_min (Fig.1b) ---
\noindent\textbf{注：}候选扫描图（Fig.~1(b)）中为便于筛选使用
\[
  v_{\max}=\frac{h_v}{\max(h_1,h_2)},
\]
而本文的 \emph{valley ratio} 定义为
\[
  v_{\min}=\frac{h_v}{\min(h_1,h_2)}.
\]
二者关系为
\[
  v_{\max}=v_{\min}\,\frac{\min(h_1,h_2)}{\max(h_1,h_2)}
  =v_{\min}\times \text{peak ratio}.
\]
所有图内标注与配置卡数值均由同一套指标计算函数输出并写入 metrics JSON，保证一致性。

\subsection{候选 B：贴边走廊 + mixed 边界}
本文将走廊贴在反射边界（$y=N$），并启用自动调参选择 $(g_x,g_y,\delta)$ 以满足双峰可视化阈值；当纯反射边界下无法满足阈值时，调参器允许 $x$ 方向采用周期边界以形成 ``绕行慢通道''。相比内部走廊构造，贴边走廊减少向外逃逸，从而显著提升 fast 权重。
\paragraph{失败对照（内部走廊）} 纯反射的内部走廊在我们的判据下给出过小的 fast 权重（早峰接近肩部噪声），因此作为失败对照保留，用以说明“贴边走廊 + 混合边界”对双峰可视化的重要性。

\paragraph{为何内部走廊早峰极小？为何贴边走廊能修复？}
内部走廊构造可视为一个 ``失败案例''：走廊处于内部带，沿 $y$ 方向的逃逸概率约为 $p_\uparrow+p_\downarrow\approx q/2$，因此 ``连续 $L$ 步留在 band'' 的概率近似为 $(1-q/2)^L$。取 $q=0.8,L=9$ 得 $(0.6)^9\approx 1\%$，与该构造的 $P_{\text{fast}}\approx 0.01$ 一致，导致早峰几乎不可见。贴边走廊方案将走廊贴在反射边界，向外逃逸被反射，仅剩单侧逃逸 $p_\uparrow\approx q(1-g_y)/4$，留在 band 的概率提升为 $(1-p_\uparrow)^L$，可达 $10\%\sim 30\%$，因此早峰从几乎不可见提升为满足可视化阈值；同时 $x$ 周期边界为慢通道提供清晰的绕行时间尺度。

  \item $N=60,\;q=0.8,\;g=(-0.25,+0.40)$，边界：$x$ 周期、$y$ 反射；start=(50,60)，target=(42,60)。
  \item 走廊：$y=60$，$x\in[43,50]$，方向 left，$\delta=0.70$，长度 $L=8$；corridor band 为 $y\in\{59,60\}$。
  \item 通道判据：命中前若未离开 corridor band 则 fast，否则 slow（与 MC 分类一致）。

\end{itemize}

\section{直观机制图解：两通道混合 + 时间尺度分离}
将轨迹按通道事件 $C\in\{\text{fast},\text{slow}\}$ 分类，则
\begin{equation}
  f(t)=\mathbb{P}(C=\text{fast})\,f(t\,|\,C=\text{fast})
  +\mathbb{P}(C=\text{slow})\,f(t\,|\,C=\text{slow}).
\end{equation}
当两类条件分布的典型时间满足 $t_{\text{fast}}\ll t_{\text{slow}}$，且 $\mathbb{P}(C=\text{fast})$ 不过小，则整体 $f(t)$ 在两峰之间形成明显 valley。

\begin{figure}[!htbp]
  \centering
  \includegraphics[width=0.65\textwidth]{\FigDir/fig3_panels/channel_cartoon.pdf}
  \caption{双峰的概念示意图：fast/slow 两通道混合 + 时间尺度分离。}
  \label{fig:channel-cartoon}
\end{figure}

\paragraph{通道判据（用于 fast/slow 分类）}
\begin{itemize}
  \item A：是否发生 $x$ 方向 wrap-around（跨越周期边界）；无 wrap 为 fast，发生 wrap 为 slow。
  \item B：是否在命中前离开 corridor band（$y\in\{59,60\}$）；保持在 band 内直到命中为 fast，离开为 slow。
  \item C：门穿越首次发生时间是否晚于 $t_v$；早穿门为 fast，晚穿门为 slow。
\end{itemize}

\paragraph{三类候选的几何解释}
\begin{itemize}
  \item A：fast 为逆漂移短路，slow 为顺漂移绕行；周期拓扑将 ``绕圈'' 变成天然慢通道。
  \item B：贴边走廊提供快通道；$x$ 周期边界 + 全局偏置提供绕行慢通道；对照扫描得到 $L_{\min}=6$。
  \item C：door + sticky 抬高穿门代价；早穿门/短路为 fast，晚穿门/绕行与 sticky 滞留为 slow；最少 local bias sites 为 $n_{\min}=0$。
\end{itemize}

候选 A、B 与 C 的 periodic 机制通过展开图展示（图~\ref{fig:A-unwrapped}、图~\ref{fig:B-unwrapped} 与图~\ref{fig:C-unwrapped}），
在两个 tile 中同时标注 target 的 image，并给出短路距离 $Dx_{\text{short}}$ 与绕行距离 $Dx_{\text{wrap}}$。

三类候选的机制共性可总结为：
\begin{itemize}
  \item A：周期拓扑提供 wrap-around 慢通道；逆偏置短路为快通道。
  \item B：贴边走廊提供快通道，$x$ 周期边界与全局偏置提供绕行慢通道。
  \item C：door + sticky 抬高穿门代价；早穿门为 fast，晚穿门（含 wrap-around 或 sticky/door 延迟）为 slow。
\end{itemize}

\section{边界条件的物理含义与保留价值}
\paragraph{reflecting} 更贴近有限容器或围墙区域的扩散/搜索问题：概率守恒且不会“漏出系统”。
\paragraph{periodic} 对应拓扑上环/环面或均匀介质的平铺近似，现实中并不常见，但可自然引入绕行慢通道，便于产生时间尺度分离（候选 A）。
\paragraph{mixed} 仅在一个方向保留 wrap-around（如圆柱表面/条带结构），兼顾可控性与理论可解析性（候选 B 与 C）。
\paragraph{选择建议} 若强调现实边界，可采用纯反射走廊基线；若强调双峰可视性，可采用贴边走廊并在 $x$ 方向引入周期边界。若强调最少 local bias，候选 A 最简但较理想化；若希望在有限墙域中仍保留可控绕行机制，候选 C 更合适。
