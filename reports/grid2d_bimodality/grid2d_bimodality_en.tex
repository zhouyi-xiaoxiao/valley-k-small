\section{Model Definition and Coordinate Convention}
The state space is $\{1,\dots,N\}\times\{1,\dots,N\}$. The walker is at $(x,y)$ and the target $(x_t,y_t)$ is absorbing. One-step transition probabilities are
\begin{align}
  p_{\text{left}} &= \frac{q}{4}(1+g_x),\quad p_{\text{right}} = \frac{q}{4}(1-g_x),\\
  p_{\text{down}} &= \frac{q}{4}(1+g_y),\quad p_{\text{up}} = \frac{q}{4}(1-g_y),\\
  p_{\text{stay}} &= 1-q.
\end{align}
We adopt the convention ``$y$ increases downward'', so $g_y>0$ means downward drift. Boundaries are reflecting, periodic, or mixed ($x$ periodic, $y$ reflecting).
\noindent From Eq.(1), \(g_x>0\) increases \(p_{\text{left}}=\frac{q}{4}(1+g_x)\) and decreases \(p_{\text{right}}=\frac{q}{4}(1-g_x)\), so we define \(g_x>0\) as a \textbf{leftward} global bias (\(g_x<0\) is rightward).

\paragraph{Local heterogeneities (combinations).}
\begin{itemize}
  \item local bias (arrow): multiply $p_{\text{stay}}$ by $(1-\delta)$ and add $\Delta=\delta p_{\text{stay}}$ to the arrow direction; other directions stay unchanged.
  \item sticky site: scale the move rate to $q_{\text{site}}=\text{factor}\cdot q$ (default 0.2); $p_{\text{stay}}$ increases accordingly.
  \item barrier: a reflecting edge adds its probability back to stay; a door passes with $p_{\text{pass}}$ and reflects with $1-p_{\text{pass}}$.
\end{itemize}

\section{Analytical derivation: from defect-free propagator to time-domain FPT}
This section provides a unified derivation chain: \emph{defect-free propagator $\to$ defect propagator $\to$ FPT generating function $\to$ AW inversion}. All three candidates fit this framework.

\subsection{Matrix form and generating function}
Let $A$ be the transition matrix for the non-absorbing system, with $A_{v,u}=\Pr(u\to v)$ so columns sum to 1 and $\mathbf{p}_{t+1}=A\mathbf{p}_t$. The propagator satisfies
\begin{equation}
  P_{n_0}(n,t)=\mathbf{e}_n^\top A^t \mathbf{e}_{n_0},
  \qquad
  \tilde{\mathbf{P}}(z)=\sum_{t\ge 0} A^t z^t=(I-zA)^{-1},
\end{equation}
so
\begin{equation}
  \tilde P_{n_0}(n,z)=\mathbf{e}_n^\top (I-zA)^{-1}\mathbf{e}_{n_0}.
\end{equation}
We now give the spectral form of $A$ and the defect update formula to obtain $\tilde P$ for any configuration.

\subsection{Spectral decomposition of the defect-free propagator}
Write the 2D update as the sum of two 1D operators (along $x$ and $y$). Let $\{\lambda_k^{(\gamma)},h_k^{(\gamma)}\}$ be eigenpairs of the 1D step matrix, where $\gamma\in\{\text{periodic},\text{reflecting}\}$ denotes boundary type. Define
\begin{equation}
  s_k^{(\gamma)} = q\,\bigl(\lambda_k^{(\gamma)}-1\bigr),\qquad
  \Lambda_{k_1,k_2}=1+\frac{s_{k_1}^{(\gamma_x)}+s_{k_2}^{(\gamma_y)}}{2}.
\end{equation}
Then the 2D defect-free propagator (Eq.(33) in PRE 102, 062124) is
\begin{equation}
  P^{(\gamma)}_{n_0}(n,t)=\sum_{k_1}\sum_{k_2} h_{k_1}^{(\gamma_x)}(x,x_0)\,h_{k_2}^{(\gamma_y)}(y,y_0)\,\Lambda_{k_1,k_2}^t.
  \label{eq:P_time}
\end{equation}
The $z$-transform is
\begin{equation}
  \tilde P^{(\gamma)}_{n_0}(n,z)=\sum_{k_1}\sum_{k_2}
  \frac{h_{k_1}^{(\gamma_x)}(x,x_0)\,h_{k_2}^{(\gamma_y)}(y,y_0)}{1-z\Lambda_{k_1,k_2}}.
  \label{eq:P_z}
\end{equation}

\paragraph{Periodic boundary (periodic)} \;\;
\begin{equation}
  \theta_k=\frac{2\pi k}{N},\quad
  \lambda_k^{(\text{per})}=\frac{1+g}{2}e^{-i\theta_k}+\frac{1-g}{2}e^{i\theta_k},\quad
  h_k^{(\text{per})}(x,x_0)=\frac{1}{N}e^{i\theta_k(x-x_0)}.
\end{equation}

\paragraph{Reflecting boundary (reflecting)} \;\;
Let $p_\ell=(1+g)/2,\;p_r=(1-g)/2,\;\rho=p_r/p_\ell$. A similarity transform symmetrizes the 1D operator, giving
\begin{equation}
  \lambda_0^{(\text{ref})}=1,\quad
  \lambda_k^{(\text{ref})}=2\sqrt{p_\ell p_r}\cos\frac{k\pi}{N}\ (k\ge 1),\quad
  h_k^{(\text{ref})}(x,x_0)=u_k(x)u_k(x_0)\sqrt{\frac{\pi_x}{\pi_{x_0}}},
\end{equation}
where $\pi_x\propto \rho^x$ is the stationary distribution and $u_k$ are orthogonal cosine modes (matching PRE Eq.(22)--(23)).

\subsection{Determinant formula for defect propagator}
Write the defect as a matrix perturbation $\underline{X}=\underline{B}-\underline{A}$. For each directed edge $(u_i\to v_i)$, define
\begin{equation}
  \eta_{u_i,v_i}=X_{v_i,u_i}=B_{v_i,u_i}-A_{v_i,u_i}.
\end{equation}
Define the difference propagator
\begin{equation}
  \tilde P^e_{\langle u-v\rangle}(n,z)=\tilde P^e_{u}(n,z)-\tilde P^e_{v}(n,z).
\end{equation}
Then the inert-heterogeneity formula (2311.00464v2) gives
\begin{equation}
  \tilde P_{n_0}(n,z)=\tilde P^e_{n_0}(n,z)-1+\frac{\det[H(n,n_0;z)]}{\det[H(z)]},
\end{equation}
with
\begin{align}
  H_{ij}(z)&=-\frac{\delta_{ij}}{z}+\eta_{u_j,v_j}\,\tilde P^e_{\langle u_j-v_j\rangle}(u_i,z)-\eta_{v_j,u_j}\,\tilde P^e_{\langle u_j-v_j\rangle}(v_i,z),\\
  H(n,n_0;z)_{ij}&=H_{ij}(z)-\tilde P^e_{\langle u_j-v_j\rangle}(n,z)\left[\eta_{u_i,v_i}\tilde P^e_{n_0}(u_i,z)-\eta_{v_i,u_i}\tilde P^e_{n_0}(v_i,z)\right].
\end{align}
In numerical implementations, we solve the equivalent linear system for stability (by Cramer equivalence).
The linear-algebra form reads
\begin{align}
  b_j(n,z)&=\tilde P^e_{u_j}(n,z)-\tilde P^e_{v_j}(n,z),\\
  a_i(n_0,z)&=\eta_{u_i,v_i}\tilde P^e_{n_0}(u_i,z)-\eta_{v_i,u_i}\tilde P^e_{n_0}(v_i,z),\\
  H_{ij}(z)&=-\frac{\delta_{ij}}{z}+\eta_{u_j,v_j}\,b_j(u_i,z)-\eta_{v_j,u_j}\,b_j(v_i,z),
\end{align}
so that
\begin{equation}
  \tilde P_{n_0}(n,z)=\tilde P^e_{n_0}(n,z)-\mathbf{b}(n,z)^\top H(z)^{-1}\mathbf{a}(n_0,z).
\end{equation}

\paragraph{Defect mapping (used in B/C)}
\begin{itemize}
  \item local bias: reduce the self-loop and add probability to the arrow direction, equivalent to $(u\to v)$ with $\eta_{u,v}=+\Delta$, plus a compensation term for $(u\to u)$ to keep column sums at 1.
  \item door/permeable barrier: crossing the door multiplies by $p_{\text{pass}}$; both $(u\to v)$ and $(v\to u)$ have $\eta_{u,v}=(p_{\text{pass}}-1)P_{u\to v}^{\text{base}}$.
  \item sticky: multiply all outgoing edges by $\alpha$; each outgoing edge $(u\to n)$ has $\eta_{u,n}=(\alpha-1)P_{u\to n}^{\text{base}}$, with a compensation term added to $(u\to u)$ to keep column sums at 1.
\end{itemize}

\subsection{FPT generating function and AW inversion}
Consider the master equation with an absorbing state:
\begin{equation}
  \mathbf{p}_{t+1}=P\,\mathbf{p}_t,\qquad \mathbf{p}_0=\mathbf{e}_{n_0},
\end{equation}
with $z$-transform $\tilde{\mathbf{p}}(z)=\sum_{t\ge 0}\mathbf{p}_t z^t$ giving
\begin{equation}
  \tilde{\mathbf{p}}(z)=(I-zP)^{-1}\mathbf{p}_0.
\end{equation}
Let the cumulative arrival probability be $F(t)=\mathbf{e}_{n_t}^\top \mathbf{p}_t$, then
\begin{equation}
  \tilde F(z)=\mathbf{e}_{n_t}^\top (I-zP)^{-1}\mathbf{p}_0.
\end{equation}
With $f(t)=F(t)-F(t-1)$ (take $F(-1)=0$), the PMF generating function is
\begin{equation}
  \tilde f(z)=(1-z)\tilde F(z)=(1-z)\,\mathbf{e}_{n_t}^\top (I-zP)^{-1}\mathbf{p}_0.
\end{equation}
We keep both notations: $\tilde F$ for the CDF generating function and $\tilde f$ for the PMF generating function (used in AW inversion). The exact recursion in time domain is
\begin{equation}
  f(t)=r^\top Q^{t-1} p_0,\quad t\ge 1,
  \label{eq:exact-recursion}
\end{equation}
so the generating function can be written as
\begin{equation}
  \tilde f(z)=\sum_{t\ge 1} f(t) z^t
  = z\,r^\top (I-zQ)^{-1} p_0.
\end{equation}
On the other hand, the renewal relation gives
\begin{equation}
  \tilde f_{n_0\to n_t}(z)=\frac{\tilde P_{n_0}(n_t,z)}{\tilde P_{n_t}(n_t,z)},
  \label{eq:Fz}
\end{equation}
which is equivalent: the former uses the transient submatrix $Q$, the latter uses the defect-free propagator.
We then apply Abate--Whitt (AW) numerical inversion to extract coefficients:
\begin{equation}
  f(t)\approx r^{-t}\frac{1}{m}\sum_{k=0}^{m-1}\tilde f(z_k)\,e^{-2\pi i k t/m},
  \qquad z_k=r\,e^{2\pi i k/m},\quad t=1,\dots,t_{\max}^{\mathrm{AW}}.
  \label{eq:AW}
\end{equation}
We choose $m=\mathrm{pow2}(\text{oversample}\cdot(t_{\max}^{\mathrm{AW}}+1))$ and set $r^m=10^{-\text{r\_pow10}}$, matching the repository defaults. Larger $r$ reduces truncation error but increases numerical sensitivity; smaller $r$ magnifies truncation error. We keep the auto-tuning strategy to balance stability and accuracy.

\begin{quote}
\textbf{Implementation workflow (Algorithm box)}
\begin{enumerate}[leftmargin=*, itemsep=0pt]
  \item Build the spectral decomposition of the defect-free $A$ (periodic/reflecting), yielding explicit mode sums for $\tilde P^e(n,z)$.
  \item Construct $H(z)$ and vectors $\mathbf{a},\mathbf{b}$ from the defect list, solve the linear system to obtain $\tilde P(n,z)$.
  \item Use the renewal formula $\tilde f(z)=\tilde P_{n_0}(n_t,z)/\tilde P_{n_t}(n_t,z)$ to get the generating function.
  \item Sample $z_k=r e^{2\pi i k/m}$ and apply AW inversion to obtain $f(t)$.
  \item Compare with exact recursion/MC, and record errors in the metrics JSON.
\end{enumerate}
\end{quote}

\section{Numerical baseline: exact/flux recursion and MC}

\paragraph{Peak and valley metrics}
For the smoothed $f(t)$, take the two dominant peak locations $t_{p1},t_{p2}$ and the minimum $t_v$ between them, and define
\begin{equation}
  h_1=f(t_{p1}),\quad h_2=f(t_{p2}),\quad
  \text{peak ratio}=\frac{\min(h_1,h_2)}{\max(h_1,h_2)},\quad
  \text{valley ratio}=\frac{f(t_v)}{\min(h_1,h_2)}.
\end{equation}
% --- clarification: v_max vs v_min (Fig.1b) ---
\noindent\textbf{Note:} For the scan in Fig.~1(b) we use
\[
  v_{\max}=\frac{h_v}{\max(h_1,h_2)},
\]
while the main-text \emph{valley ratio} is
\[
  v_{\min}=\frac{h_v}{\min(h_1,h_2)}.
\]
The relation is
\[
  v_{\max}=v_{\min}\,\frac{\min(h_1,h_2)}{\max(h_1,h_2)}
  =v_{\min}\times \text{peak ratio}.
\]

\subsection{Candidate B: edge corridor + mixed boundary}
This study places the corridor on the reflecting edge ($y=N$) and enables auto-tuning of $(g_x,g_y,\delta)$ to meet the bimodality visibility threshold. If a pure reflecting boundary cannot satisfy the threshold, the tuner allows an $x$-periodic boundary to create a ``wrap-around slow channel''. Compared with an interior corridor, the edge corridor reduces escape and boosts the fast weight.
\paragraph{Failure baseline (interior corridor)} A pure reflecting interior corridor yields too small a fast weight (the early peak becomes a shoulder), and is kept as a failure baseline to illustrate why ``edge corridor + mixed boundary'' is needed for visible bimodality.

\paragraph{Why is the interior-corridor early peak tiny? Why does an edge corridor fix it?}
The interior-corridor construction behaves as a ``failure case'': the corridor lies in an interior band, and escape in $y$ occurs with probability $p_\uparrow+p_\downarrow\approx q/2$, so the probability of staying in the band for $L$ steps is roughly $(1-q/2)^L$. With $q=0.8,L=9$, this gives $(0.6)^9\approx 1\%$, matching $P_{\text{fast}}\approx 0.01$ for that construction and making the early peak nearly invisible. The edge-corridor variant places the corridor on a reflecting boundary, so outward escape is reflected and only one-sided escape remains ($p_\uparrow\approx q(1-g_y)/4$), raising the stay probability to $(1-p_\uparrow)^L$, which can reach $10\%$--$30\%$. This turns the early peak from ``almost invisible'' into ``visibly bimodal''; the $x$-periodic boundary then provides a clear wrap-around time scale for the slow channel.

\paragraph{Configuration Card}
\begin{itemize}
  \item $N=60,\;q=0.8,\;g=(-0.25,+0.40)$, boundary: $x$ periodic, $y$ reflecting; start=(50,60), target=(42,60).
  \item Corridor: $y=60$, $x\in[43,50]$, direction left, $\delta=0.70$, length $L=8$; corridor band $y\in\{59,60\}$.
  \item Channel criterion: stay in the corridor band until hitting the target is fast; leaving the band before hitting is slow (matches MC classification).
  \item Peak/valley metrics: $t_{p1}=33,\;t_v=220,\;t_{p2}=473$; $h_2/h_1=0.989$; valley ratio$=6.04\times 10^{-2}$.
\end{itemize}

\subsection{Candidate C: door + sticky + wrap-around channel}
We place a permeable door ($p_{\text{pass}}=0.2$) on a vertical barrier and add a $3\times 3$ sticky block (factor=0.2) to its left. With $x$-periodic boundary and leftward global bias ($g_x=+0.5$), we classify channels by first door-crossing time: early crossing is fast, late crossing (including wrap-around and sticky/door delays) is slow; bimodality already appears at $n_{\text{bias}}=0$.

\section{Intuitive mechanism diagram: two-channel mixture + time-scale separation}
Classify trajectories by channel event $C\in\{\text{fast},\text{slow}\}$, then
\begin{equation}
  f(t)=\mathbb{P}(C=\text{fast})\,f(t\,|\,C=\text{fast})
  +\mathbb{P}(C=\text{slow})\,f(t\,|\,C=\text{slow}).
\end{equation}
When the typical times satisfy $t_{\text{fast}}\ll t_{\text{slow}}$ and $\mathbb{P}(C=\text{fast})$ is not too small, a visible valley forms between the peaks.

\begin{figure}[!htbp]
  \centering
  \includegraphics[width=0.65\textwidth]{\FigDir/fig3_panels/channel_cartoon.pdf}
  \caption{Conceptual cartoon: bimodality from two-channel mixture + time-scale separation.}
  \label{fig:channel-cartoon}
\end{figure}

\paragraph{Channel criteria (for fast/slow classification)}
\begin{itemize}
  \item A: whether $x$ wrap-around occurs; no wrap is fast, any wrap is slow.
  \item B: whether the trajectory leaves the corridor band ($y\in\{59,60\}$) before hitting; staying in the band is fast, leaving is slow.
  \item C: whether the first door-crossing time is later than $t_v$; early crossing is fast, late crossing is slow.
\end{itemize}

\paragraph{Geometric interpretation of the three candidates}
\begin{itemize}
  \item A: fast is an against-drift shortcut; slow is a with-drift wrap-around; periodic topology makes the slow channel natural.
  \item B: the edge corridor provides the fast channel; the $x$-periodic boundary plus global bias provides the wrap-around slow channel; scan yields $L_{\min}=6$.
  \item C: door + sticky raise the cost of slow paths; early door crossing is fast, late door crossing with wrap-around or sticky delays is slow; the minimum local bias count is $n_{\min}=0$.
\end{itemize}

The shared mechanism can be summarized as follows:
\begin{itemize}
  \item A: periodic topology provides a wrap-around slow channel; the against-bias shortcut yields the fast channel.
  \item B: the edge corridor provides the fast channel; the $x$-periodic boundary and global bias provide the wrap-around slow channel.
  \item C: door + sticky raise the door-crossing cost; early door crossing is fast, late crossing (including wrap-around or sticky/door delays) is slow.
\end{itemize}

\section{Boundary conditions: physical meaning and rationale}
\paragraph{reflecting} Closest to diffusion/search in finite containers or walled domains: probability is conserved and does not ``leak'' out of the system.
\paragraph{periodic} Corresponds to a ring/torus topology or a tiled homogeneous medium; less common in reality but naturally introduces a wrap-around slow channel (Candidate A).
\paragraph{mixed} Periodic in one direction only (e.g., cylindrical surfaces or strip geometries), balancing realism and analytical control (Candidates B and C).
\paragraph{Recommendation} If realistic boundaries are emphasized, consider a pure reflecting corridor baseline; if bimodality visibility is prioritized, use an edge corridor and allow an $x$-periodic boundary. If minimal local bias is a priority, Candidate A is simplest but more idealized; if one wants a controllable wrap-around mechanism within walled domains, Candidate C is more suitable.
