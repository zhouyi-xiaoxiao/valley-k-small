\section{模型定义与坐标约定}
状态空间为 $\{1,\dots,N\}\times\{1,\dots,N\}$。记当前位置为 $(x,y)$，目标点 $(x_t,y_t)$ 为吸收态。每步更新为
\begin{align}
  p_{\text{left}} &= \frac{q}{4}(1+g_x),\quad p_{\text{right}} = \frac{q}{4}(1-g_x),\\
  p_{\text{down}} &= \frac{q}{4}(1+g_y),\quad p_{\text{up}} = \frac{q}{4}(1-g_y),\\
  p_{\text{stay}} &= 1-q.
\end{align}
本文采用 ``$y$ 轴向下为正'' 的坐标约定，因此 $g_y>0$ 表示向下偏置。本文仅使用\textbf{全反射边界}：越界概率全部加回 $p_{\text{stay}}$。
\noindent 由式(1)可见 $g_x>0$ 会增大 $p_{\text{left}}$ 并减小 $p_{\text{right}}$，因此本文约定 $g_x>0$ 表示\textbf{向左}全局偏置（$g_x<0$ 表示向右偏置）。

\paragraph{局部异质性（组合）}
\begin{itemize}
  \item local bias（箭头）：将 $p_{\text{stay}}$ 乘 $(1-\delta)$，并把 $\Delta=\delta p_{\text{stay}}$ 加到箭头方向；其余方向不变。
  \item sticky site：将该格点的移动概率缩放为 $q_{\text{site}}=\text{factor}\cdot q$，$p_{\text{stay}}$ 相应上升。
  \item barrier：反射墙将该方向概率加回 stay；半透门以 $p_{\text{pass}}$ 通过、$1-p_{\text{pass}}$ 反射。
\end{itemize}

\section{FPT 基础：定义、直观与三种计算方法}
\paragraph{定义与直观}
设首达时间（first-passage time）为
\[
  \tau=\inf\{t\ge 1:\,X_t=(x_t,y_t)\},
\]
其 PMF 为 $f(t)=\Pr(\tau=t)$，CDF 为 $F(t)=\Pr(\tau\le t)$，生存概率为
\[
  S(t)=\Pr(\tau>t)=1-F(t).
\]
直观上，$f(t)$ 描述“恰好在第 $t$ 步首次抵达目标”的概率，因此图中的双峰就是 $f(t)$ 在早期与晚期各有一个显著局部极大值。对于吸收边界模型，$F(t)$ 也可以理解为“累计吸收概率”，而 $S(t)$ 则是“尚未吸收的剩余概率质量”。

\paragraph{三种计算方法（直观版）}
本文的 $f(t)$ 由三种方法得到，并在图中叠加对比：
\begin{itemize}
  \item \textbf{Exact recursion（确定性递推）}：把所有非吸收态组成 transient 子矩阵 $Q$，一步一步推进概率分布：
  \[
    p_{t+1}=Qp_t,\qquad f(t)=r^\top p_t,\quad t\ge 1,
  \]
  其中 $r$ 是从 transient 进入吸收态的“流出概率”。这是“逐步推演”的确定性算法，数值误差仅来自浮点与截断。
  \item \textbf{AW 反演（生成函数数值反演）}：先构造生成函数 $\tilde f(z)$（本报告用 propagator + 缺陷公式得到），再用 Abate--Whitt 反演
  \[
    f(t)\approx r^{-t}\frac{1}{m}\sum_{k=0}^{m-1}\tilde f(z_k)e^{-2\pi i k t/m},
  \]
  得到整段时间序列。直观上是“先到 $z$ 域算，再回到时间域”。
  \item \textbf{MC（蒙特卡洛）}：随机模拟大量轨迹，统计首达时间直方图得到 $f(t)$ 近似。它最灵活，但尾部噪声大，需要更多样本。
\end{itemize}

\paragraph{AW 与 Exact 的对比（给新手）}
\begin{itemize}
  \item \textbf{准确性}：Exact 是确定性的基准；AW 是高精度数值反演，参数选得合适时可与 Exact 几乎重合；MC 为统计估计，误差随样本数 $\propto 1/\sqrt{N}$ 衰减。
  \item \textbf{速度与适用场景}：Exact 适合中等规模、需要高精度的 $t_{\max}$；AW 在“已知生成函数”的前提下可同时得到大量 $t$ 值，但每个 $z_k$ 需要求解线性系统，缺陷数大时会变慢；MC 适合快速验证与复杂几何，但需要更多样本才能看清尾部与双峰细节。
  \item \textbf{本报告用法}：Exact 作为基准，AW 用于与 Exact 对齐验证，MC 用来展示通道分类与统计直观。
\end{itemize}

\section{解析推导：从无缺陷 propagator 到时域 FPT}
本节给出统一推导链条：\emph{无缺陷 propagator $\to$ 有缺陷 propagator $\to$ FPT 生成函数 $\to$ AW 反演}。本文构造均可落在同一框架内。

\subsection{矩阵形式与生成函数}
设无吸收系统的转移矩阵为 $A$，约定 $A_{v,u}=\Pr(u\to v)$，因此列和为 1，且 $\mathbf{p}_{t+1}=A\mathbf{p}_t$。传播子满足
\begin{equation}
  P_{n_0}(n,t)=\mathbf{e}_n^\top A^t \mathbf{e}_{n_0},
  \qquad
  \tilde{\mathbf{P}}(z)=\sum_{t\ge 0} A^t z^t=(I-zA)^{-1},
\end{equation}
从而
\begin{equation}
  \tilde P_{n_0}(n,z)=\mathbf{e}_n^\top (I-zA)^{-1}\mathbf{e}_{n_0}.
\end{equation}
下面给出 $A$ 的谱分解形式与缺陷更新公式，以便对任意构造得到 $\tilde P$。

\subsection{无缺陷 propagator 的谱分解（反射边界）}
将二维转移写成两个一维算子叠加（沿 $x$ 与 $y$ 方向）。设一维步进矩阵的特征对为 $\{\lambda_k^{(\text{ref})},h_k^{(\text{ref})}\}$。定义
\begin{equation}
  s_k^{(\text{ref})} = q\,\bigl(\lambda_k^{(\text{ref})}-1\bigr),\qquad
  \Lambda_{k_1,k_2}=1+\frac{s_{k_1}^{(\text{ref})}+s_{k_2}^{(\text{ref})}}{2}.
\end{equation}
则二维无缺陷 propagator（PRE 102, 062124 的 Eq.(33) 形式）为
\begin{equation}
  P^{(\text{ref})}_{n_0}(n,t)=\sum_{k_1}\sum_{k_2} h_{k_1}^{(\text{ref})}(x,x_0)\,h_{k_2}^{(\text{ref})}(y,y_0)\,\Lambda_{k_1,k_2}^t.
  \label{eq:P_time}
\end{equation}
对时间做 $z$ 变换得到
\begin{equation}
  \tilde P^{(\text{ref})}_{n_0}(n,z)=\sum_{k_1}\sum_{k_2}
  \frac{h_{k_1}^{(\text{ref})}(x,x_0)\,h_{k_2}^{(\text{ref})}(y,y_0)}{1-z\Lambda_{k_1,k_2}}.
  \label{eq:P_z}
\end{equation}

\paragraph{反射边界（reflecting）} \;\;
设 $p_\ell=(1+g)/2,\;p_r=(1-g)/2,\;\rho=p_r/p_\ell$。可通过相似变换将一维算子对称化，得到
\begin{equation}
  \lambda_0^{(\text{ref})}=1,\quad
  \lambda_k^{(\text{ref})}=2\sqrt{p_\ell p_r}\cos\frac{k\pi}{N}\ (k\ge 1),\quad
  h_k^{(\text{ref})}(x,x_0)=u_k(x)u_k(x_0)\sqrt{\frac{\pi_x}{\pi_{x_0}}},
\end{equation}
其中 $\pi_x\propto \rho^x$ 为平衡分布，$u_k$ 为正交余弦模（与 PRE Eq.(22)-(23) 一致）。

\subsection{有缺陷 propagator 的 determinant 公式}
将缺陷写为矩阵扰动 $\underline{X}=\underline{B}-\underline{A}$。对每条有向边 $(u_i\to v_i)$ 定义
\begin{equation}
  \eta_{u_i,v_i}=X_{v_i,u_i}=B_{v_i,u_i}-A_{v_i,u_i}.
\end{equation}
并定义差分 propagator
\begin{equation}
  \tilde P^e_{\langle u-v\rangle}(n,z)=\tilde P^e_{u}(n,z)-\tilde P^e_{v}(n,z).
\end{equation}
则 2311.00464v2 的 inert-heterogeneity 公式给出
\begin{equation}
  \tilde P_{n_0}(n,z)=\tilde P^e_{n_0}(n,z)-1+\frac{\det[H(n,n_0;z)]}{\det[H(z)]},
\end{equation}
其中
\begin{align}
  H_{ij}(z)&=-\frac{\delta_{ij}}{z}+\eta_{u_j,v_j}\,\tilde P^e_{\langle u_j-v_j\rangle}(u_i,z)-\eta_{v_j,u_j}\,\tilde P^e_{\langle u_j-v_j\rangle}(v_i,z),\\
  H(n,n_0;z)_{ij}&=H_{ij}(z)-\tilde P^e_{\langle u_j-v_j\rangle}(n,z)\left[\eta_{u_i,v_i}\tilde P^e_{n_0}(u_i,z)-\eta_{v_i,u_i}\tilde P^e_{n_0}(v_i,z)\right].
\end{align}
数值实现中，我们等价地求解线性系统得到 $\tilde P$（由 Cramer 规则与上式等价），以获得更好的数值稳定性。
亦可将其写成线性代数形式：
\begin{align}
  b_j(n,z)&=\tilde P^e_{u_j}(n,z)-\tilde P^e_{v_j}(n,z),\\
  a_i(n_0,z)&=\eta_{u_i,v_i}\tilde P^e_{n_0}(u_i,z)-\eta_{v_i,u_i}\tilde P^e_{n_0}(v_i,z),\\
  H_{ij}(z)&=-\frac{\delta_{ij}}{z}+\eta_{u_j,v_j}\,b_j(u_i,z)-\eta_{v_j,u_j}\,b_j(v_i,z),
\end{align}
从而
\begin{equation}
  \tilde P_{n_0}(n,z)=\tilde P^e_{n_0}(n,z)-\mathbf{b}(n,z)^\top H(z)^{-1}\mathbf{a}(n_0,z).
\end{equation}

\paragraph{缺陷映射}
\begin{itemize}
  \item local bias：减少 self-loop 并把概率加到箭头方向，等价于 $(u\to v)$ 的 $\eta_{u,v}=+\Delta$，并对 $(u\to u)$ 添加补偿项以保持列和为 1。
  \item door/permeable barrier：跨门通过概率乘 $p_{\text{pass}}$，对应边 $(u\to v)$ 与 $(v\to u)$ 具有 $\eta_{u,v}=(p_{\text{pass}}-1)P_{u\to v}^{\text{base}}$。
  \item sticky：所有出边概率乘 $\alpha$，每条出边 $(u\to n)$ 的 $\eta_{u,n}=(\alpha-1)P_{u\to n}^{\text{base}}$，并对 $(u\to u)$ 添加补偿项以保持列和为 1。
\end{itemize}

\subsection{FPT 生成函数与 AW 反演}
考虑包含吸收态的主方程
\begin{equation}
  \mathbf{p}_{t+1}=P\,\mathbf{p}_t,\qquad \mathbf{p}_0=\mathbf{e}_{n_0},
\end{equation}
对时间做 $z$ 变换 $\tilde{\mathbf{p}}(z)=\sum_{t\ge 0}\mathbf{p}_t z^t$ 得
\begin{equation}
  \tilde{\mathbf{p}}(z)=(I-zP)^{-1}\mathbf{p}_0.
\end{equation}
记累积到达概率 $F(t)=\mathbf{e}_{n_t}^\top \mathbf{p}_t$，则
\begin{equation}
  \tilde F(z)=\mathbf{e}_{n_t}^\top (I-zP)^{-1}\mathbf{p}_0.
\end{equation}
由 $f(t)=F(t)-F(t-1)$（取 $F(-1)=0$）得到 PMF 的生成函数
\begin{equation}
  \tilde f(z)=(1-z)\tilde F(z)=(1-z)\,\mathbf{e}_{n_t}^\top (I-zP)^{-1}\mathbf{p}_0.
\end{equation}
本文保留两者记号：$\tilde F$ 为 CDF 的生成函数，$\tilde f$ 为 PMF 的生成函数（AW 反演使用 $\tilde f$）。
Exact recursion 的时域表达为
\begin{equation}
  f(t)=r^\top Q^{t-1} p_0,\quad t\ge 1,
  \label{eq:exact-recursion}
\end{equation}
因此生成函数可写为
\begin{equation}
  \tilde f(z)=\sum_{t\ge 1} f(t) z^t
  = z\,r^\top (I-zQ)^{-1} p_0.
\end{equation}
另一方面，由 renewal 关系亦有
\begin{equation}
  \tilde f_{n_0\to n_t}(z)=\frac{\tilde P_{n_0}(n_t,z)}{\tilde P_{n_t}(n_t,z)},
  \label{eq:Fz}
\end{equation}
两者等价：前者以 transient 子矩阵 $Q$ 表达，后者以无吸收 propagator 表达。
随后采用 Abate--Whitt (AW) 数值反演将 $z$ 域系数化：
\begin{equation}
  f(t)\approx r^{-t}\frac{1}{m}\sum_{k=0}^{m-1}\tilde f(z_k)\,e^{-2\pi i k t/m},
  \qquad z_k=r\,e^{2\pi i k/m},\quad t=1,\dots,t_{\max}^{\mathrm{AW}}.
  \label{eq:AW}
\end{equation}
本文选取 $m=\mathrm{pow2}(\text{oversample}\cdot(t_{\max}^{\mathrm{AW}}+1))$，并设 $r^m=10^{-\text{r\_pow10}}$，与仓库内 AW 设置保持一致。
其中 $r$ 越接近 1，截断误差越小但数值更敏感；$r$ 过小则会放大截断误差，本文沿用自动选参策略以平衡稳定性与精度。

\begin{quote}
\textbf{实现流程（Algorithm box）}
\begin{enumerate}[leftmargin=*, itemsep=0pt]
  \item 构建无缺陷 $A$ 的谱分解（反射边界），得到 $\tilde P^e(n,z)$ 的显式模态和。
  \item 由缺陷列表构造 $H(z)$ 与向量 $\mathbf{a},\mathbf{b}$，解线性系统得到 $\tilde P(n,z)$。
  \item 用 renewal 公式 $\tilde f(z)=\tilde P_{n_0}(n_t,z)/\tilde P_{n_t}(n_t,z)$ 得到生成函数。
  \item 对 $z_k=r e^{2\pi i k/m}$ 采样并用 AW 反演得到 $f(t)$。
  \item 与 exact recursion/MC 对比误差，并记录为指标 JSON。
\end{enumerate}
\end{quote}

\subsection{AW vs exact recursion：原理与速度对比}
\paragraph{方法原理（含公式）}
设 $Q$ 为 transient 子矩阵，$r$ 为从 transient 进入吸收态的概率向量，$p_0$ 为起点分布。
Exact recursion 通过时域递推
\[
  p_{t+1}=Qp_t,\qquad f(t)=r^\top p_t,\quad t\ge 1,
\]
并可写为闭式
\[
  f(t)=r^\top Q^{t-1}p_0,\qquad F(t)=1-\mathbf{1}^\top Q^t p_0.
\]
因此在给定 $t_{\max}$ 时得到确定性 PMF（数值误差仅来自浮点运算与截断）。AW 则先在 $z$ 域构造生成函数
\[
  \tilde f(z)=z\,r^\top (I-zQ)^{-1}p_0
  =\frac{\tilde P_{n_0}(n_t,z)}{\tilde P_{n_t}(n_t,z)},
\]
再由 Cauchy 积分提取系数：
\[
  f(t)=\frac{1}{2\pi i}\oint_{|z|=r}\frac{\tilde f(z)}{z^{t+1}}\,dz
  \approx r^{-t}\frac{1}{m}\sum_{k=0}^{m-1}\tilde f(z_k)e^{-2\pi i k t/m}.
\]

\paragraph{复杂度与适用性}
Exact recursion 的时间复杂度为 $O(T|E|)$（$T=t_{\max}$，$|E|$ 为非零转移数），内存 $O(|E|)$，对大多数中等尺度参数扫描更稳健。AW 每个 $z$ 需解 $M\times M$ 线性系统（$M$ 为 defect 数），总体 $O(m M^3)$，FFT 的 $O(m\log m)$ 通常不占主导；当 $M$ 很小时，AW 在长时间窗口上更具优势。AW 的 $z_k$ 采样天然可并行，而 Exact recursion 迭代性强但实现简单、数值稳定。

\paragraph{准确性说明}
Exact recursion 在给定 $t_{\max}$ 内是“确定性精确”的（仅有浮点舍入误差），本报告以其作为基准。AW 属于数值反演，误差来自有限 $m$ 的离散化、$r$ 的选择以及 $\tilde f(z)$ 线性求解的数值条件；增大 $m$ 或调整 $r$ 可系统降低误差。本报告中 AW 与 exact 的 $L^1$ 误差约 $10^{-14}$（表~\ref{tab:aw-errors}），因此两者在数值精度意义下“同等准确”。

\paragraph{速度对比设置}
选取本报告双峰构造 R1、R7、C3、R6、NB4、NB5 作为速度基准，使用各自 $t_{\max}$；AW 取 oversample=4、r\_pow10=12，对应 $m=8192/16384/65536/131072$。为避免完整 AW 过慢，先在 $n_z=32$ 个 $z_k$ 上计时得到单点耗时，再估计
\[
  t_{\mathrm{AW}}\approx t_{\mathrm{setup}} + (m/n_z)\,t_{\mathrm{eval}} + t_{\mathrm{FFT}}.
\]
同时加入低缺陷对照 S0（全网格无内墙，仅 1 个 sticky 与 1 个 door，$M=9$），可完整测量 AW 全流程，用于展示 AW 的可行区间。其参数如下：
S0 的参数为 $N=30$、start=(1,15)、target=(30,15)、$g_x=-0.4$，并设置 sticky$(15,15,\alpha=0.1)$ 与 door$(20,15)\leftrightarrow(21,15)$、$p_{\text{pass}}=0.05$。
计时脚本为 \texttt{code/benchmark\_aw\_exact.py}（单线程），结果见表~\ref{tab:aw-speed}。

\paragraph{结论（速度优先）}
对走廊/门构造（$M\approx 7.5\times 10^2$--$10^3$），Exact recursion 在 $t_{\max}\le 2\times 10^4$ 内用时 $0.06$--$0.70$ s；AW 估计总耗时为 5--280 min，瓶颈来自每个 $z$ 的线性求解。即便 R7（$M=754$）和 C3（$M=1004$）这样的 ``小走廊''，AW 估计总耗时仍达 5.2 min 与 24.7 min，但 AW 反演本身仅 $0.23$--$4.76$ ms。S0（$M=9$）可完整测量 AW 全流程并进入秒级（1.44 s），但仍慢于 Exact（0.056 s）；若缺陷更少或可并行/加速线性求解，AW 才可能占优。总体上，若需参数扫描或中等 $t_{\max}$，Exact recursion 更稳健；若缺陷极少且需要长尾，AW 更具优势。

\section{数值基线：Exact/flux 递推与 MC}
Exact 递推在 transient 空间计算 $f(t)=r^\top Q^{t-1} p_0$（式~\ref{eq:exact-recursion}），作为确定性基准，并与 AW 反演及 MC 直方图叠加对比（图~\ref{fig:R1-fpt}、\ref{fig:R7-fpt}、\ref{fig:C3-fpt}、\ref{fig:R6-fpt}、\ref{fig:NB4-fpt}、\ref{fig:NB5-fpt}、\ref{fig:S1-fpt}、\ref{fig:S2-fpt}）。MC 轨迹同时按 ``快/慢通道'' 分类，以解释双峰来源。

\paragraph{峰谷指标定义}
对平滑后的 $f(t)$，取两个局部主峰位置 $t_{p1},t_{p2}$ 及其间最小值 $t_v$，并定义
\begin{equation}
  h_1=f(t_{p1}),\quad h_2=f(t_{p2}),\quad
  \text{peak ratio}=\frac{\min(h_1,h_2)}{\max(h_1,h_2)},\quad
  \text{valley ratio}=\frac{f(t_v)}{\min(h_1,h_2)}.
\end{equation}
平滑窗口采用 9，峰谷指标与通道权重均由统一脚本输出并写入 metrics JSON，保证一致性。

\section{直观机制图解：两通道混合 + 时间尺度分离}
将轨迹按通道事件 $C\in\{\text{fast},\text{slow}\}$ 分类，则
\begin{equation}
  f(t)=\mathbb{P}(C=\text{fast})\,f(t\,|\,C=\text{fast})
  +\mathbb{P}(C=\text{slow})\,f(t\,|\,C=\text{slow}).
\end{equation}
当两类条件分布的典型时间满足 $t_{\text{fast}}\ll t_{\text{slow}}$，且 $\mathbb{P}(C=\text{fast})$ 不过小，则整体 $f(t)$ 在两峰之间形成明显 valley。

\begin{figure}[!htbp]
  \centering
  \includegraphics[width=0.65\textwidth]{\FigDir/fig3_panels/channel_cartoon.pdf}
  \caption{双峰的概念示意图：fast/slow 两通道混合 + 时间尺度分离。}
  \label{fig:channel-cartoon}
\end{figure}

\paragraph{通道判据（用于 fast/slow 分类）}
\begin{itemize}
  \item R1：命中前始终在主走廊（$y=30$）为 fast；进入 U 形绕行为 slow。
  \item R7：命中前始终在主走廊（$y=30$）为 fast；进入上方绕行支路为 slow。
  \item C3：命中前始终在主动轨道（$y=29$）为 fast；进入 detour 回路为 slow。
  \item R6：主要沿 $y=20$ 车道为 fast；主要沿 $y=40$ door 车道为 slow。
  \item NB4：命中前未到达顶边 $y=1$ 为 fast；到达过顶边则为 slow（对应顺时针绕行）。
  \item NB5：判据与 NB4 相同；全局偏置仅改变 fast/slow 权重与峰位。
  \item S1：命中前未到达顶边 $y=1$ 段为 fast；到达顶边段则为 slow（软通道绕行）。
  \item S2：命中前未到达 $y=10$ 段为 fast；到达 $y=10$ 段则为 slow（内层软绕行）。
\end{itemize}

\paragraph{构造的几何/动力学解释}
\begin{itemize}
  \item R1：反射墙将空间压缩为准 1D 走廊；短直走廊为 fast，长绕行为 slow。
  \item R7：主走廊内的半透门提供快捷通道；绕行支路形成更长的几何路程，实现孔道 vs 绕行的时间尺度分离。
  \item C3：强 local bias 轨道提供“主动运输”快通道；错过入口后进入长回路形成慢通道。
  \item R6：两车道几何长度相近，但 door 阵列显著降低有效通行率，形成动力学慢通道。
  \item NB4：外边界顺时针 bias 形成“传送带”，逆向短路与顺时针绕行形成深双峰。
  \item NB5：在 NB4 传送带上叠加全局偏置，进一步倾向顺时针路径，使谷更深。
  \item S1/S2：通过 sticky 将通道内的横向逃逸压到很小，再用 local bias 将 stay 质量定向释放，从而在无内部墙的情况下形成准 1D 软通道与清晰的 fast/slow 分离。
\end{itemize}
