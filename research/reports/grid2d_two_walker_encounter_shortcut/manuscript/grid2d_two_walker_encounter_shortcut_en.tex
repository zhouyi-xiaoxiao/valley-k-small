\begin{abstract}
This extension delivers two implementation-ready blocks.
First, we provide a robust numerical routine to verify the equivalence between appendix Eq.~(A1) and Eq.~(A8) in the ``Product of two generating functions'' section, including contour selection, adaptive discretization, and diagnostics.
Second, we construct a 2D two-walker encounter model with directed shortcuts and show a shortcut-controlled transition from single peak to double peak in encounter first-passage distributions.
All figures/tables/data are generated by one script and are reproducible.
\end{abstract}

\section{Problem Statement}
We target two tasks:
\begin{enumerate}[leftmargin=1.4em]
  \item a numerically stable verification of
  \[
  \frac{1}{2\pi i}\oint_C\frac{du}{u}\,\widetilde P_{n_0}(n,u)\widetilde P_{m_0}\!\left(n,\frac{z}{u}\right)
  =
  \sum_{k_1,k_2=0}^{N-1}\frac{h_{k_1}(n,n_0)h_{k_2}(n,m_0)}{1-z(1+s_{k_1})(1+s_{k_2})},
  \]
  i.e. Eq.~(A1)$\Leftrightarrow$(A8) under valid contour conditions;
  \item a 2D two-walker encounter construction under shortcuts where the encounter FPT changes from single-peak to double-peak.
\end{enumerate}

\section{A1 $\rightarrow$ A8: Compact Derivation Roadmap}
Let $\alpha_k=1+s_k=1+q(\cos(2\pi k/N)-1)$ and
\[
\widetilde P_{n_0}(n,u)=\sum_{k=0}^{N-1}\frac{h_k(n,n_0)}{1-u\alpha_k}.
\]
Insert into Eq.~(A1):
\[
I(z)=\frac{1}{2\pi i}\oint_C\sum_{k_1,k_2}
\frac{h_{k_1}(n,n_0)h_{k_2}(n,m_0)}{(1-u\alpha_{k_1})(u-z\alpha_{k_2})}\,du.
\]
For each $(k_1,k_2)$, poles are at $u=1/\alpha_{k_1}$ (outer family) and $u=z\alpha_{k_2}$ (inner family).
Choose a circle $|u|=r$ such that
\[
\max_k|z\alpha_k|<r<\min_{k:\alpha_k\neq 0}\left|\frac{1}{\alpha_k}\right|.
\]
Then only the inner poles contribute, yielding residues
\[
\operatorname*{Res}_{u=z\alpha_{k_2}}=
\frac{h_{k_1}(n,n_0)h_{k_2}(n,m_0)}{1-z\alpha_{k_1}\alpha_{k_2}},
\]
and summing residues gives Eq.~(A8).

\section{Numerical Routine (Implementation-Ready)}
Parameterize $u=re^{i\theta}$ and use $du/u=i\,d\theta$:
\[
I(z)=\frac{1}{2\pi}\int_0^{2\pi}
\widetilde P_{n_0}(n,re^{i\theta})
\widetilde P_{m_0}\!\left(n,\frac{z}{re^{i\theta}}\right)
\,d\theta.
\]
We evaluate this by trapezoidal rule:
\[
I_M=\frac1M\sum_{j=0}^{M-1}F(r e^{2\pi i j/M}).
\]

\paragraph{Stability rules used in code.}
\begin{enumerate}[leftmargin=1.4em]
  \item \textbf{Contour feasibility check:} enforce $r_{\mathrm{in}}<r<r_{\mathrm{out}}$ with
  $r_{\mathrm{in}}=\max_k|z\alpha_k|$,
  $r_{\mathrm{out}}=\min_{\alpha_k\neq 0}|1/\alpha_k|$.
  \item \textbf{Radius choice:} default $r=\sqrt{r_{\mathrm{in}}r_{\mathrm{out}}}$.
  \item \textbf{Adaptive refinement:} start from $M=16$ and double until
  $|I_{2M}-I_M|<\varepsilon(1+|I_{2M}|)$.
  \item \textbf{Cross-check:} compare contour LHS against direct RHS Eq.~(A8).
\end{enumerate}

\section{2D Two-Walker Encounter Model With Shortcut}
State space is a reflecting $N\times N$ grid for each walker; one-step transitions are lazy biased moves plus an optional directed shortcut from source to destination.
We compute first encounter at any site:
\[
T_{\mathrm{enc}}=\min\{t\ge1: X_t=Y_t\},
\qquad
f_{\mathrm{enc}}(t)=\mathbb P(T_{\mathrm{enc}}=t).
\]

\paragraph{Exact time-domain evolution.}
Let $J_t(i,j)=\mathbb P(X_t=i,Y_t=j,\,T_{\mathrm{enc}}>t)$.
Independent propagation gives
\[
J_{t+1}=P_1^\top J_t P_2,
\]
encounter flux is diagonal mass
\[
f_{\mathrm{enc}}(t+1)=\mathrm{tr}(J_{t+1}),
\]
then enforce absorption by zeroing the diagonal of $J_{t+1}$.
This avoids Monte Carlo noise and is fast for moderate grids.

\paragraph{Shortcut channel decomposition.}
For walker A, we split contributions into trajectories that have used the shortcut vs never used it, by propagating two coupled tensors $(J_t^{(0)},J_t^{(1)})$.
This yields interpretable components ``shortcut-used'' and ``no-shortcut'' in $f_{\mathrm{enc}}(t)$.

\section{Pitfalls and Debug Checklist}
\begin{enumerate}[leftmargin=1.4em]
  \item \textbf{Contour crossing poles (A1/A8):} if $r_{\mathrm{in}}\ge r_{\mathrm{out}}$, do not integrate.
  \item \textbf{Near-singular denominators:} monitor $\min_{k_1,k_2}|1-z\alpha_{k_1}\alpha_{k_2}|$.
  \item \textbf{False micro-peaks in encounter FPT:} we ignore ultra-early transients with $t<t_{\mathrm{ignore}}$ and apply consistent smoothing.
  \item \textbf{Mass balance:} always track $\sum_t f(t)$ and survival tail at $t_{\max}$.
\end{enumerate}
